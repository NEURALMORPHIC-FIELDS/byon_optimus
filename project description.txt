## Quick Start

python
from fhrss_fcpe_unified import UnifiedFHRSS_FCPE, UnifiedConfig

# Initialize system
system = UnifiedFHRSS_FCPE()

# Add context (text or embeddings)
from sentence_transformers import SentenceTransformer
model = SentenceTransformer('all-MiniLM-L6-v2')

text = "Machine learning is a subset of artificial intelligence..."
embedding = model.encode(text)

ctx_id = system.encode_context(
    embedding.reshape(1, -1),
    metadata={'text': text}
)

# Retrieve similar contexts
query_emb = model.encode("What is AI?")
results = system.retrieve_similar(query_emb, top_k=5)

# Test recovery from corruption
recovery = system.test_recovery(ctx_id, loss_percent=0.30)
print(f"Recovery: {recovery['cosine_similarity']:.4f}")  # 1.0000

import json, hashlib
from nacl.signing import SigningKey, VerifyKey
from nacl.encoding import HexEncoder

def canonical_json(obj):
    return json.dumps(obj, sort_keys=True, separators=(",", ":"), ensure_ascii=False)

def sha256_hex(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8")).hexdigest()

def sign(payload: dict, signing_key_hex: str) -> str:
    sk = SigningKey(signing_key_hex, encoder=HexEncoder)
    msg = canonical_json(payload).encode("utf-8")
    return sk.sign(msg).signature.hex()

def verify(payload: dict, signature_hex: str, verify_key_hex: str) -> bool:
    vk = VerifyKey(verify_key_hex, encoder=HexEncoder)
    msg = canonical_json(payload).encode("utf-8")
    vk.verify(msg, bytes.fromhex(signature_hex))
    return True

  { id: "empathy_1", re: /\b(îmi pare rău|imi pare rau|te înțeleg|te inteleg|înteleg|simt că|simt ca)\b/i },
  { id: "story_1", re: /\b(odată|imaginează-ți|imagineaza-ti|hai să-ți spun o poveste)\b/i },
  { id: "therapy_1", re: /\b(traumă|trauma|vindecare|coping|anxietate|anxiety)\b/i },
  { id: "meta_1", re: /\b(ca model|ca asistent|nu pot|nu sunt capabil)\b/i } // elimină scuze/meta
];

// reguli de formă
function computeStyleScore(content: string): { score: number; violations: string[] } {
  const violations: string[] = [];

  // forbidden patterns
  for (const p of FORBIDDEN_PATTERNS) {
    if (p.re.test(content)) violations.push(p.id);
  }

  // penalizează prolixitate
  const len = content.trim().length;
  if (len > 3500) violations.push("too_long");

  // penalizează lipsa de structură (minim: 3 linii)
  const lines = content.split("\n").filter(l => l.trim().length > 0);
  if (lines.length < 3) violations.push("too_unstructured");

  // score simplu: 100 - penalizări
  let score = 100;
  score -= violations.filter(v => v.startsWith("empathy")).length * 25;
  score -= violations.filter(v => v.startsWith("story")).length * 25;
  score -= violations.filter(v => v.startsWith("therapy")).length * 15;
  score -= violations.filter(v => v.startsWith("meta")).length * 10;
  if (violations.includes("too_long")) score -= 10;
  if (violations.includes("too_unstructured")) score -= 10;

  if (score < 0) score = 0;
  return { score, violations };
}

function ajvErrorsToStrings(errs: ErrorObject[] | null | undefined): string[] {
  if (!errs?.length) return [];
  return errs.map(e => {
    const path = e.instancePath || "(root)";
    const msg = e.message || "schema error";
    return `${path}: ${msg}`;
  });
}

export class ByonValidator {
  private ajv: Ajv;
  private validateFn: (data: unknown) => boolean;
  private schemaId = "byon://schema/style-contract/v1";

  constructor(schema: object) {
    this.ajv = new Ajv({ allErrors: true, strict: true });
    addFormats(this.ajv);

    // Register schema under its $id
    this.ajv.addSchema(schema, this.schemaId);
    const fn = this.ajv.getSchema(this.schemaId);
    if (!fn) throw new Error("BYON schema not registered");
    this.validateFn = (data: unknown) => fn(data) as boolean;
  }

  validate(doc: any, minScore = 85): ByonValidationResult {
    const okSchema = this.validateFn(doc);
    const schemaErrors = ajvErrorsToStrings((this.ajv.getSchema(this.schemaId) as any)?.errors);

    // content scoring
    const content = String(doc?.output?.content ?? "");
    const { score, violations } = computeStyleScore(content);

    const ok = Boolean(okSchema) && score >= minScore;

    const errors: string[] = [];
    if (!okSchema) errors.push(...schemaErrors);
    if (score < minScore) errors.push(`style_score_below_threshold: ${score} < ${minScore}`);
    for (const v of violations) errors.push(`violation:${v}`);

    return { ok, errors, score, violations };
  }
}

import schema from "./byon_contract.schema.json";
import { ByonValidator, ByonValidationResult } from "./byon_validator";

export type RegenContext = {
  attempt: number;
  lastErrors: string[];
  lastScore: number;
  lastViolations: string[];
};

export type RegenerateFn<TDoc> = (ctx: RegenContext) => Promise<TDoc>;

export type ValidateOrRegenerateOptions = {
  minScore?: number;        // default 85
  maxAttempts?: number;     // default 4
  hardFail?: boolean;       // default true: aruncă eroare după maxAttempts
};

export type ValidateOrRegenerateResult<TDoc> = {
  ok: boolean;
  attempts: number;
  doc: TDoc;
  validation: ByonValidationResult;
  history: Array<{ attempt: number; score: number; errors: string[]; violations: string[] }>;
};

const validator = new ByonValidator(schema as any);

export async function validate_or_regenerate<TDoc extends any>(
  initialDoc: TDoc,
  regenerate: RegenerateFn<TDoc>,
  opts: ValidateOrRegenerateOptions = {}
): Promise<ValidateOrRegenerateResult<TDoc>> {
  const minScore = opts.minScore ?? 85;
  const maxAttempts = opts.maxAttempts ?? 4;
  const hardFail = opts.hardFail ?? true;

  const history: ValidateOrRegenerateResult<TDoc>["history"] = [];

  let doc = initialDoc;
  let validation = validator.validate(doc, minScore);

  history.push({
    attempt: 0,
    score: validation.score,
    errors: validation.errors,
    violations: validation.violations
  });

  if (validation.ok) {
    return { ok: true, attempts: 0, doc, validation, history };
  }

  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    const ctx: RegenContext = {
      attempt,
      lastErrors: validation.errors,
      lastScore: validation.score,
      lastViolations: validation.violations
    };

    doc = await regenerate(ctx);
    validation = validator.validate(doc, minScore);

    history.push({
      attempt,
      score: validation.score,
      errors: validation.errors,
      violations: validation.violations
    });

    if (validation.ok) {
      return { ok: true, attempts: attempt, doc, validation, history };
    }
  }

  const result: ValidateOrRegenerateResult<TDoc> = {
    ok: false,
    attempts: maxAttempts,
    doc,
    validation,
    history
  };

  if (hardFail) {
    const last = history[history.length - 1];
    const msg =
      `BYON validation failed after ${maxAttempts} attempts.\n` +
      `last_score=${last.score} min_score=${minScore}\n` +
      `violations=${last.violations.join(",")}\n` +
      `errors:\n- ${last.errors.join("\n- ")}`;
    throw new Error(msg);
  }

  return result;
}
